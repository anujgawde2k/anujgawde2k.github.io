//constructor overloading

class Student{
private int sid;
private String sname;
public Student(int a,String b){
sid=a;
sname=b;
}
public Student(String b,int a){
sname=b;
sid=a;

}
public void getData(){
System.out.println("\nStudent id : "+sid);
System.out.println("Name \t\t : "+sname);
}
}

class consover{
public static void main(String [] args){
Student s1=new Student(3248,"varsha jha");
Student s2=new Student("abhishek jha",6452);
s1.getData();
s2.getData();
}
}
